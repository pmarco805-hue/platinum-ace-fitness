function q(s,el=document){return el.querySelector(s)}
function qa(s,el=document){return Array.from(el.querySelectorAll(s))}
function weekIndex(){
  const now=new Date(), onejan=new Date(now.getFullYear(),0,1);
  return Math.ceil((((now-onejan)/86400000)+onejan.getDay()+1)/7);
}
function currentQuote(dayIndex){return QUOTES[(weekIndex()+dayIndex)%QUOTES.length]}

function load(k, d){try{return JSON.parse(localStorage.getItem(k))??d}catch(e){return d}}
function save(k, v){localStorage.setItem(k, JSON.stringify(v))}

let currentDay=0;
function setDay(i){
  currentDay=i;
  qa('nav button').forEach((b,idx)=>b.classList.toggle('active', idx===i));
  renderDay(i);
}
function renderDay(i){
  const names=Object.keys(WORKOUTS), name=names[i];
  q('#day-title').textContent=name;
  q('#quote').textContent='‚Äú'+currentQuote(i)+'‚Äù';
  const wrap=q('#exercise-list'); wrap.innerHTML='';
  const checks=load(CHECKS_KEY,{});

  WORKOUTS[name].forEach(([label,url])=>{
    const id=name+'::'+label, checked=!!checks[id];
    const row=document.createElement('div'); row.className='exercise';

    const left=document.createElement('label'); left.style.display='flex'; left.style.alignItems='center'; left.style.gap='8px';
    const cb=document.createElement('input'); cb.type='checkbox'; cb.className='checkbox'; cb.checked=checked;
    cb.addEventListener('change',()=>{const d=load(CHECKS_KEY,{}); d[id]=cb.checked; save(CHECKS_KEY,d)});
    const text=document.createElement('span'); text.textContent=label;
    left.appendChild(cb); left.appendChild(text);

    const link=document.createElement('a'); link.href=url; link.target='_blank'; link.textContent='Video';
    row.appendChild(left); row.appendChild(link); wrap.appendChild(row);
  });
}
function resetChecks(){localStorage.removeItem(CHECKS_KEY); renderDay(currentDay)}

let timerInterval=null, remaining=0;
function startTimer(sec){
  clearInterval(timerInterval); remaining=sec; q('#timer-display').textContent=formatTime(remaining);
  timerInterval=setInterval(()=>{
    remaining-=1;
    if(remaining<=0){clearInterval(timerInterval); remaining=0; q('#timer-display').textContent='Done'; try{navigator.vibrate&&navigator.vibrate([150,80,150])}catch(e){}}
    else q('#timer-display').textContent=formatTime(remaining);
  },1000);
}
function formatTime(s){const m=Math.floor(s/60), ss=s%60; return String(m).padStart(2,'0')+':'+String(ss).padStart(2,'0')}

function loadLogs(){return load(LOGS_KEY,[])}
function saveLogs(v){save(LOGS_KEY,v)}
function addLog(){
  const w=parseFloat(q('#log-weight').value||''), ww=parseFloat(q('#log-waist').value||'');
  if(isNaN(w)&&isNaN(ww)){alert('Enter weight and/or waist.');return}
  const entry={date:new Date().toISOString(), weight:w, waist:ww};
  const logs=loadLogs(); logs.unshift(entry); saveLogs(logs.slice(0,20));
  q('#log-weight').value=''; q('#log-waist').value=''; renderProgress();
}
function renderProgress(){
  const logs=loadLogs(), list=q('#progress-logs'); list.innerHTML='';
  logs.slice(0,6).forEach(item=>{
    const row=document.createElement('div'); row.className='progress-entry';
    const date=new Date(item.date).toLocaleDateString();
    row.innerHTML=`<div class='label'>${date}</div><div>Weight: <b>${item.weight||'-'}</b></div><div>Waist: <b>${item.waist||'-'}</b></div>`;
    list.appendChild(row);
  });
  const photos=load(PHOTOS_KEY,[]), gal=q('#photo-gallery'); gal.innerHTML='';
  photos.forEach((p,idx)=>{
    const box=document.createElement('div');
    const img=document.createElement('img'); img.src=p.data; img.alt='progress photo';
    const del=document.createElement('button'); del.className='btn'; del.textContent='Delete'; del.style.marginTop='4px';
    del.addEventListener('click',()=>{const arr=load(PHOTOS_KEY,[]); arr.splice(idx,1); save(PHOTOS_KEY,arr); renderProgress()});
    box.appendChild(img); box.appendChild(del); gal.appendChild(box);
  });
}
function addPhoto(file){
  const reader=new FileReader();
  reader.onload=()=>{const arr=load(PHOTOS_KEY,[]); arr.unshift({data:reader.result, date:new Date().toISOString()}); save(PHOTOS_KEY,arr.slice(0,40)); renderProgress()};
  reader.readAsDataURL(file);
}
function showTab(which){
  q('#screen-workouts').style.display=(which==='workouts')?'block':'none';
  q('#screen-progress').style.display=(which==='progress')?'block':'none';
}
